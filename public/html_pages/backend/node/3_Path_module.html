Path module
Usage
Syntax
const path = require("path");

console.log(__filename);   //full path to the file
console.log(__dirname);   //string to the folder of the file

console.log(path.basename(__filename));   //returns just the file name
console.log(path.basename(__dirname));   //returns just the folder name

console.log(path.extname(__filename));   //returns just the file extension
console.log(path.extname(__dirname));   //returns just the folder extension

console.log(path.parse(__filename));   //returns object with info

console.log(path.isAbsolute(__filename));   //returns true if the path is absolute(from root)

console.log(path.join("folder1", "index.html"));   //joins the path

console.log(path.resolve("folder1", "index.html"));   //joins the path to current path
console.log(path.resolve("/folder1", "index.html"));   //doesnt join the path
