const fs = require("fs");

Reading a file asynchronously
Syntax
fs.readFile( filename, encoding, callback_function )
Code
const fs = require("fs");

const fileContents = fs.readFile("./file.txt", "utf-8", (error, data) => {
    if(error){
        console.log(error);
    }
    else{
        console.log(data);
    }
});
Output

Writing to a file asynchronously
Syntax
fs.writeFile( file, data, options, callback )
Code
const fs = require("fs");

fs.writeFile("./greet.txt", "Hello world!", {flag: "a"}, (err) => {
    if(err){
        console.log(err)
    }
    else{
        console.log("success");
    }
});

fs promises
Syntax
fsPromises.readFile( path, options )
Code
const fs = require("fs/promises");

//then() -> catch() version
fs.readFile("file.txt", "utf-8")
    .then(data => console.log(data))
    .catch((error) => console.log(error))

//async -> await version
async function readFile() {
    try{
        await data = await fs.readFile("file.txt", "utf-8");
        console.log(data);
    } catch(err){
        console.log(err);
    }
}

readFile();
