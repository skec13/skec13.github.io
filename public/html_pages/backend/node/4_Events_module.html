Events module
Usage
Syntax
const EventEmitter = require("events");

const emiter = new EventEmitter();

emitter.emit("newEvent", "argument");   //fire an "newEvent" event with arguments

emitter.on("newEvent", (argument) => {
   console.log("event received" + argument);
});
Extending from EventEmitter
Syntax
class PizzaShop extends EventEmitter {
    constructor() {
        super();
    }

    order() {
        this.emit("order");
    }
}
