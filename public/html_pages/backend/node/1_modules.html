Custom modules
Syntax
//externalModule.js
const add = (a, b) => {
    return a + b;
};

const subtract = (a, b) => {
    return a - b;
};

module.exports = {
    add,
    subtract
}

//Main.js
const math = require('./externalModule'); //import external module
console.log(math.add(4, 5)); //function from external module
ES modules
Syntax
//externalModule.mjs  (.mjs file extension, change type in packet.json)
const add = (a, b) => {
    return a + b;
};

const subtract = (a, b) => {
    return a - b;
};

export default {
    add,
    subtract
}

//Main.mjs
import math from './externalModule.mjs';
console.log(math.add(4, 5));
