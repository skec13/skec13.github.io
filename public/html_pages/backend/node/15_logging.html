Logging
Morgan
Syntax
const express = require('express');
const morgan = require('morgan');

const app = express();
const port = 5000;

//morgan middleware
app.use(morgan('tiny'));

app.get('/', (req, res) => {
    res.send('Hello');
})

app.listen(port, () => console.info(`App listening on port ${port}`))

Custom tokens
Syntax
const express = require('express');
const morgan = require('morgan');

const app = express();
const port = 5000;

//morgan middleware
morgan.token('type', function(req, res){
    return req.headers['content-type']
});
app.use(morgan(':method :url :status :type'));

app.get('/', (req, res) => {
    res.send('Hello');
})

app.listen(port, () => console.info(`App listening on port ${port}`))
Save logs to a file
Syntax
const express = require('express');
const morgan = require('morgan');
const path = require('path');
const fs = require('fs');

const app = express();
const port = 5000;

const accessLogStream = fs.createWriteStream(path.join(__dirname, 'access.log'), { flags: 'a' });

//morgan middleware
morgan.token('type', function(req, res){
    return req.headers['content-type']
});
app.use(morgan(':method :url :status :type', { stream: accessLogStream }));

app.get('/', (req, res) => {
    res.send('Hello');
})

app.listen(port, () => console.info(`App listening on port ${port}`))
