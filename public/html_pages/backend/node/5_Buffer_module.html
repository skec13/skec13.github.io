Buffer module
Buffer.alloc()
Syntax
Buffer.alloc(size, fill, encoding);   //creates a new buffer of any size
Code
const buf = Buffer.alloc(6);

console.log("Buf: " + buf);
Output
buffer.write()
Syntax
buffer.write(value, start, bytes, encoding);   //writes a string to the buffer
Code
const buf = Buffer.from('abcdef');

buf.write('qq',2);

console.log(buf.toString());
Output
Buffer.byteLength()
Syntax
Buffer.byteLength(string, encoding);   //length of a buffer object
Code
var buf = Buffer.alloc(15);

var len = Buffer.byteLength(buf);

console.log(len);
Output
Buffer.compare()
Syntax
Buffer.compare(buf1, buf2);  //compares two buffers
Code
var buf1 = Buffer.from('abc');
var buf2 = Buffer.from('abc');
var x = Buffer.compare(buf1, buf2);
console.log(x);
Output
Buffer.concat()
Syntax
Buffer.concat(arr, length);   //joins the array into one buffer
Code
var buf1 = Buffer.from('a');
var buf2 = Buffer.from('b');
var buf3 = Buffer.from('c');
var arr = [buf1, buf2, buf3];

var buf = Buffer.concat(arr);
console.log(buf);
Output
buffer.fill()
Syntax
buffer.fill(value, start, end, encoding);  //fills a buffer with the specified values
Code
var buf = Buffer.alloc(15);

buf.fill('a', 5, 10);

console.log(buf);
Output
Buffer.from()
Syntax
Buffer.from(obj, encoding);   //creates a Buffer object from an object(string/array/buffer)
Code
var buf = Buffer.from('abc');

console.log(buf);
Output
buffer.entries()
Syntax
buffer.entries();   //used to loop over a buffer
Code
var buf = Buffer.from('abc');

for (x of buf.entries()) {
  console.log(x);
}
Output
buffer.slice()
Syntax
buffer.slice(start, end);   //creates a new array from selected elements in an array
Code
var x = Buffer.from('abcdef');

var y = x.slice(2,5);

console.log(y.toString());
Output
