Http module
Web server
Syntax
const http = require('http');

const server = http.createServer((res, req) => {
    res.writeHead(200, { "Content-Type": "text/plain" });
    res.end("Hello World!");
});

server.listen(3000, () => {
    console.log('Server running on port 3000');
});
JSON Response
Syntax
const http = require('http');

const server = http.createServer((res, req) => {
    const customObj = {
        name: "Matevz",
    }

    res.writeHead(200, { "Content-Type": "application/json" });   //JSON header
    res.end(JSON.stringify(customObj));   //convert object to JSON
});

server.listen(3000, () => {
    console.log('Server running on port 3000');
});
HTML Response
Syntax
const http = require('http');
const fs = require('fs');

const server = http.createServer((res, req) => {
    res.writeHead(200, { "Content-Type": "text/html" });   //html header
    fs.createReadStream("./index.html").pipe(res);  //HTML from file
});

server.listen(3000, () => {
    console.log('Server running on port 3000');
});
Basic routing
Syntax
const http = require('http');
const fs = require('fs');

const server = http.createServer((res, req) => {
    if(res.url === "/") {
        res.writeHead(200, {"Content-Type": "text/plain"});
        res.end("Home Page");
    } else if (res.url === "/about"){
        res.writeHead(200, {"Content-Type": "text/plain"});
        res.end("About Page");
    } else if (res.url === "/api"){
        res.writeHead(200, {"Content-Type": "text/json"});
        res.end(JSON.stringify(customObj));
    } else {
        res.writeHead(404);
        res.end("Page not found");
    }
});

server.listen(3000, () => {
    console.log('Server running on port 3000');
});
