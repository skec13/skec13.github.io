Syntax
<?php
    /*Logical operators*/
    $x and $y  //or &&
    $x or $y  //or ||
    $x xor $y

    /*String operators*/
    $txt1 . $txt2   //concatenation
    $txt1 .= $txt2   //concatenation assignment (concats and adds one string to another)

    /*Array operators*/
    $x + $y   //union of both arrays
    $x == $y	//returns true if arrays have the same key/value pairs
    $x === $y   //returns true if arrays have same key/value pairs in the same order and same type
    $x != $y	
    $x !== $y	

    /*Conditional assignment operators*/
    $x = expr1 ? expr2 : expr3   //x is expr2 if expr1 is true, otherwise x is expr3
    $x = expr1 ?? expr2   //x is expr1 if expr1 exists and is not null, otherwise x is expr2


?>
