Magic methods
__get()
Syntax
<?php
/*if we try to assign value to an undefined or inaccesable property we can implement the --get()
  to return something even if the value isnt set or is private
*/

//Before.php

class Invoice {

}

$invoice -> amount;   //returns undefined property, because Invoice class hasnt implemented this property


//Solution.php

class Invoice {
    public function __get(string $name){
        var_dump($name);
    }
}

$invoice -> amount;   //returns string(6) "amount"
?>
