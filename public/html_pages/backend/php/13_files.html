Filesystem
Syntax
<?php
$dir = scandir(__DIR__);   /*returns files in the current directory*/
is_dir($dir[0]);   /*checks if the first element is a directory*/
is_file($dir[0]);   /*checks if the first element is a file*/

mkdir(path, mode, recursive, context);   /*creates a directory*/
rmdir(dir, context);   /*removes an empty directory*/

file_exists(path);   /*checks if file exists*/
filesize(filename);   /*returns file size*/
clearstatcache();   /*call before checking filesize for correct result*/

feof(file);   /*checks if the end of the file has been reached*/
?>

fopen(), fread(), fclose()
Syntax
<?php
fopen(filename, mode, include_path, context);   /*opens a file or URL and returns a resource data*/
fread(file, length);   /*reads from an open file*/
fclose(filepointer);   /*closes an open file pointer*/
?>
Code

<!DOCTYPE >
<html lang="en">
<body>
    
<?php
    $myfile = fopen("webdictionary.txt", "r") or die("Unable to open file!");
    echo fread($myfile,filesize("webdictionary.txt"));
    fclose($myfile);
?> 

</body>
</html>
Output
fgets(), fgetcsv(), fgetc()
Syntax
<?php
fgets(file, length);   /*returns a line from an open file*/
fgetcsv(file, length, separator, enclosure);   /*returns a line from CSV file*/
fgetc(file);   /*returns a single character from an open file*/
?>
Code

<!DOCTYPE >
<html lang="en">
<body>
    
<?php
    $myfile = fopen("webdictionary.txt", "r") or die("Unable to open file!");
    echo fgets($myfile);
    fclose($myfile);
?> 

<?php
    $myfile = fopen("webdictionary.txt", "r") or die("Unable to open file!");
    // Output one line until end-of-file
    while(!feof($myfile)) {
    echo fgets($myfile) . "<br>";
    }
    fclose($myfile);
?> 

</body>
</html>

fopen(), fwrite()
Syntax
<?php
fopen(filename, mode, include_path, context);   /*opens a file or URL and returns a resource data*/
fwrite(file, string, length);   /*writes to an open file*/
?>
Code

<!DOCTYPE >
<html lang="en">
<body>
    
<?php
    //Overwriting
    $myfile = fopen("newfile.txt", "w") or die("Unable to open file!");
    $txt = "John Doe\n";
    fwrite($myfile, $txt);
    $txt = "Jane Doe\n";
    fwrite($myfile, $txt);
    fclose($myfile);
?> 

<?php
    //Appending
    $myfile = fopen("newfile.txt", "a") or die("Unable to open file!");
    $txt = "Donald Duck\n";
    fwrite($myfile, $txt);
    $txt = "Goofy Goof\n";
    fwrite($myfile, $txt);
    fclose($myfile);
?> 

</body>
</html>
