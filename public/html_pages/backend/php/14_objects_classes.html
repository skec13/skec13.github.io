Objects and classes
Define class
Syntax
<?php
class Fruit {
    // Properties
    public $name;
    public $color;
    
    // Methods
    function set_name($name) {
        $this->name = $name;
    }
    function get_name() {
        return $this->name;
    }
}
?>
Define object
Syntax
<?php
    $apple = new Fruit();
    $banana = new Fruit();
    $apple->set_name('Apple');   //uses method to set name
    $apple->name = 'Apple';   //changes the property value without method
?>

Class constants
Syntax
<?php
class Goodbye {
    const LEAVING_MESSAGE = "Thank you for visiting W3Schools.com!";
}
    
echo Goodbye::LEAVING_MESSAGE;   //access constant from outside the class
?>

<?php
class Goodbye {
    const LEAVING_MESSAGE = "Thank you for visiting W3Schools.com!";
    public function byebye() {
        echo self::LEAVING_MESSAGE;
    }
}
  
$goodbye = new Goodbye();
$goodbye->byebye();   //access the contant with a method
?>
Class inheritance
Syntax
<?php
class Fruit {
    public $name;
    public $color;
    public function __construct($name, $color) {
        $this->name = $name;
        $this->color = $color;
    }
    public function intro() {
        echo "The fruit is {$this->name} and the color is {$this->color}.";
    }
}
    
// Strawberry is inherited from Fruit
class Strawberry extends Fruit {
    public function message() {
        echo "Am I a fruit or a berry? ";
    }
}

$strawberry = new Strawberry("Strawberry", "red");
$strawberry->message();
$strawberry->intro();
?>

Static properties(can be called without creating an instrance of a class)
Syntax
<?php
class pi {
    public static $value = 3.14159;
}
    
echo pi::$value;  //echo static property
?>


<?php
class pi {
    public static $value=3.14159;
    public function staticValue() {
        return self::$value;   //static method can be accessed by self keyword
    }
}
  
$pi = new pi();
echo $pi->staticValue();
?>


<?php
class pi {
    public static $value=3.14159;
}
  
class x extends pi {
    public function xStatic() {
        return parent::$value;   //use parent keyword to access static property from parent
    }
}
?>
Static methods
Syntax
<?php
class greeting {
    public static function welcome() {
        echo "Hello World!";
    }
}
    
greeting::welcome();   //calls static method
?>
