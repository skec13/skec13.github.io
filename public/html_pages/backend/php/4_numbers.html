Number types
Syntax
<?php
    /*Integers*/
    PHP_INT_MAX   //the largest integer supported
    PHP_INT_MIN   //the smallest integer supported
    PHP_INT_SIZE   //the size of an integer in bytes 

    is_int();   //checks if type is integer

    /*Floats*/
    PHP_FLOAT_MAX   //the largest representable floating point number
    PHP_FLOAT_MIN   //the smallest representable positive floating point number
    PHP_FLOAT_DIG   //the number of decimal digits that can be rounded into a float and back without precision loss
    PHP_FLOAT_EPSILON   //the smallest representable positive number x, so that x + 1.0 != 1.0

    is_float();   //checks if type is float

    /*Infinity-larger than PHP_FLOAT_MAX*/
    is_finite();   //checks if a value is finite
    is_infinite();

    /*NaN*/
    is_nan();   //checks if a value is not a number

    /*Numerical strings*/
    is_numeric();   //returns true if variable is a number or a numeric strings

    /*Converting casting strings and floats to integers*/
    $x = "23465.768";
    $int_cast = (int)$x;   //removes decimal part

    $x = 234.56;
    $int_cast1 = (int)$x;

?>
