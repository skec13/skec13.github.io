<?php
    function addNumbers(int $a, int $b) {
        return $a + $b;
    }

    echo addNumbers(4,5);
    echo addNumbers(4,"5")   //strict is not enabled so string is converted to int

    /*strict mode*/
    declare(strict_types=1);

    function addNumbers(int $a, int $b) {
    return $a + $b;
    }
    
    echo addNumbers(5, "5 days");   //error

    /*default argument value*/
    function number(int $a = 5){
        return a++;
    }

    /*return type declarations*/
    declare(strict_types=1); // strict requirement

    function addNumbers(float $a, float $b) : int {
        return (int)($a + $b);
    }
    echo addNumbers(1.2, 5.2); 

    /*passing argument by reference*/
    function add_five(&$value) {
        $value += 5;
    }
      
    $num = 2;
    add_five($num);   //adds 5 to num2 because of the &(reference) in arguments
    echo $num;

    /*multiple arguments*/
    function sum(...$numbers){
        $sum = 0;
        foreach($numbers as num){
            $sum += $number;
        }
        return $sum;
    }

    $arr = [1,2,3,4];
    sum(...$arr);   //you can pass array into function with spread operator

    /*named arguments-PHP8*/
    function divide(int $x, int $y){
        return $x / $y;
    }

    $first = 1;
    $second = 5;
    divide(y: $second, x: $first);   //you can change the order of parameters and it still works

?>
