Dates
Date and time
Syntax
<?php
date(format, timestamp);   //returns date
date_default_timezone_set("America/New_York");   /*sets timezone*/
date_default_timezone_get();   /*returns current timezone*/

mktime(hour, minute, second, month, day, year);   /*returns the Unix timestamp for a date*/

strtotime(time, now);   /*string to Unix timestamp*/

?>
Code

<!DOCTYPE >
<html lang="en">
<body>
    
<?php
    echo "1: " .date('m/d/Y g:ia') . '<br/>';
    echo "2: " .date('h:i:sa') . '<br/>';
    echo "3: " .date('m/d/Y g:ia', mktime(0,0,0,4,10,null)) . '<br/>';
    echo "4: " .date('m/d/Y g:ia', strtotime('2021-01-18 07:00:00')) . '<br/>';
    echo "5: " .date('m/d/Y g:ia', strtotime('tommorow')) . '<br/>';
    echo "6: " .date('m/d/Y g:ia', strtotime('first day of february')) . '<br/>';
    echo "7: " .date('m/d/Y g:ia', strtotime('second friday of January')) . '<br/>';
?> 




</body>
</html>
Output
DateTime object
Syntax
<?php
$dateTime = new DateTime('05/12/2021 3:30PM', new DateTimeZone('Europe/Amsterdam'));

$dateTime->setDate(2021, 4, 21);   //changes the date
$dateTime->setTime(2, 15);   //changes time
?>

<?php
$date = '15/05/2021 15:30PM';
$dateTime = DateTime::createFromFormat('d/m/Y g:iA', $date);
?>

<?php
$dateTime1 = new DateTime('15/05/2021 15:30PM');
$dateTime2 = new DateTime('15/05/2021 19:30PM');

$difference = $dateTime1->diff($dateTime2);   //calculates the difference between dates
?>   

<?php
$dateTime = new DateTime('15/05/2021 15:30PM');
$interval = new DateInterval('P3M2D');

$dateTime->add( $interval);
?>  

<?php
$period = new DatePeriod(new DateTime('05/01/2021'), new DateInterval('P1D'), new DateTime('05/31/2021'));

foreach($period as $date){
    echo $date->format('m/d/Y');
}
?>
