Cookies
Send a cookie
Syntax
import express from "express";

const app = express();

app.get("/", (req, res) => {
    //set a cookie
    res.cookie('name', 'matevz', { maxAge: 60000 });
    res.sendStatus(200);
})

app.listen(PORT, () => {
    console.log(`http://localhost:${PORT}`);
}
Receive a cookie
Syntax
import express from "express";
import cookieParse from "cookie-parser";

const app = express();

//enable cookie-parser middleware
app.use(cookieParser());

app.get("/", (req, res) => {
    //get a cookie from request object
    const cookie = request.cookies.name;
    res.sendStatus(200);
})

app.listen(PORT, () => {
    console.log(`http://localhost:${PORT}`);
}
Receive a signed cookie
Syntax
import express from "express";
import cookieParse from "cookie-parser";

const app = express();

//enable cookie-parser middleware
app.use(cookieParser("secretKey"));

app.get("/", (req, res) => {
    //get a cookie from request object
    const cookie = request.signedCookies.name;
    res.sendStatus(200);
})

app.listen(PORT, () => {
    console.log(`http://localhost:${PORT}`);
}
