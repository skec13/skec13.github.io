Integration and E2E testing
supertest-setup
Syntax
npm install supertest

e2e folder

index.test.js file extension

//package.json
"scripts": {
    "test:e2e": "jest --testPathPattern=src/e2e"
}
index.test.js
Syntax
import request from "supertest";
import express from 'express';

const app = express();

app.get("/hello", (req, res) => res.sendStatus(200).send({ msg: "name" }));

//testing
describe('hello endpoint', () => {
    it('get /hello endpoint and expect 200', async () => {
        const response = await request(app).get("/hello");
        expect(response.statusCode).toBe(200);
        expect(response.body).toEqual({ msg: "name" });
    })
})
