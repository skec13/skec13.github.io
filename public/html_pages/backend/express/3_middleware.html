Middleware
Basic global middleware
Syntax
const express = require('express');
const app = express();

const PORT = process.env.PORT || 3000;

const loggingMiddleware = (req, res, next) => {
    console.log(`${req.method}-${req.url}`);
    next();
}

//use middleware on all routes
app.use(loggingMiddleware);

app.listen(PORT, () => {
    console.log(`http://localhost:${PORT}`);
}
Basic route middleware
Syntax
const express = require('express');
const app = express();

const PORT = process.env.PORT || 3000;

const loggingMiddleware = (req, res, next) => {
    console.log(`${req.method}-${req.url}`);
    next();
}

//use middleware on specific route
app.get("/", loggingMiddleware, (req, res) => {
    res.status(200);
)

app.listen(PORT, () => {
    console.log(`http://localhost:${PORT}`);
}
Refactor the put request
Syntax
const express = require('express');
const app = express();

const PORT = process.env.PORT || 3000;

const resolveIndexByUserId = (req, res, next) => {
    const { params: { id } } = req;
    const parsedId = parseInt(id);

    //check if legit number
    if(isNan(parseId)) return res.status(400).send("Bad request");

    //find user in mockUsers with the same id as in params
    const findUserIndex = mockUsers.findIndex((user) => user.id === parsedId);

    //check if user exists
    if(!findUserIndex === -1) return res.sendStatus(404);

    //passing findUserIndex to the next middleware
    req.findUserIndex = findUserIndex;

    next();
}

//use middleware on specific route
app.put("/", resolveIndexByUserId, (req, res) => {
    //destructuring body and findUserIndex from middleware
    const { body, findUserIndex } = req;

    mockUsers[findUserIndex] = { id: mockUsers[findUserIndex].id, ...body };

    return res.sendStatus(200);
)

app.listen(PORT, () => {
    console.log(`http://localhost:${PORT}`);
}
Morgan middleware
Syntax
const express = require('express');
const morgan = require("morgan");

const app = express();

const PORT = process.env.PORT || 3000;

app.use(morgan('dev'));   //enable morgan logging

app.get("/", (req, res) => {
    res.sendStatus(200);
)

app.listen(PORT, () => {
    console.log(`http://localhost:${PORT}`);
}
