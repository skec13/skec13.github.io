INNER JOIN clause
Syntax
SELECT column_name(s)
FROM table1
INNER JOIN table2
ON table1.column_name = table2.column_name;

-- The INNER JOIN keyword returns only rows with a match in both tables.
Code
SELECT ProductID, ProductName, CategoryName
FROM Products
INNER JOIN Categories ON Products.CategoryID = Categories.CategoryID;
Output
LEFT JOIN clause
Syntax
SELECT column_name(s)
FROM table1
LEFT JOIN table2
ON table1.column_name = table2.column_name;

--  The LEFT JOIN keyword returns all records from the left table, even if there are no matches in the right table.
RIGHT JOIN clause
Syntax
SELECT column_name(s)
FROM table1
RIGHT JOIN table2
ON table1.column_name = table2.column_name; 

--  The RIGHT JOIN keyword returns all records from the right table, and the matching records from the left table.
FULL JOIN clause
Syntax
SELECT column_name(s)
FROM table1
FULL OUTER JOIN table2
ON table1.column_name = table2.column_name
WHERE condition;

--  The FULL OUTER JOIN keyword returns all records when there is a match in left or right table records.
