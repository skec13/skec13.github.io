ANY and ALL
ANY Operator
Syntax
SELECT column_name(s)
FROM table_name
WHERE column_name operator ANY
  (SELECT column_name
  FROM table_name
  WHERE condition);

-- ANY means that the condition will be true if the operation is true for any of the values in the range.
Code
SELECT ProductName
FROM Products
WHERE ProductID = ANY
  (SELECT ProductID
  FROM OrderDetails
  WHERE Quantity = 10);

/*The following SQL statement lists the ProductName
if it finds ANY records in the OrderDetails table
has Quantity equal to 10 (this will return TRUE because
the Quantity column has some values of 10)*/
Output
ALL Operator
Syntax
SELECT column_name(s)
FROM table_name
WHERE column_name operator ALL
  (SELECT column_name
  FROM table_name
  WHERE condition);

-- ALL means that the condition will be true only if the operation is true for all values in the range.
Code
SELECT ProductName
FROM Products
WHERE ProductID = ALL
  (SELECT ProductID
  FROM OrderDetails
  WHERE Quantity = 10);

/*The following SQL statement lists the ProductName
if ALL the records in the OrderDetails table has Quantity equal to 10.
This will of course return FALSE because the Quantity column
has many different values (not only the value of 10)*/
