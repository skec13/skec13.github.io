Transition classes
Syntax
<div class="transition-none"></div>   <!-- transition-property: none; -->
<div class="transition-all"></div>   <!-- transition-property: all; -->
                                     <!-- transition-timing-function: cubic-bezier(0.4, 0, 0.2, 1); -->
                                     <!-- transition-duration: 150ms; -->
<div class="transition"></div>   <!-- transition-property: color, background-color, border-color, text-decoration-color, fill, stroke, opacity, box-shadow, transform, filter, backdrop-filter; -->
                                 <!-- transition-timing-function: cubic-bezier(0.4, 0, 0.2, 1); -->
                                 <!-- transition-duration: 150ms; -->
<div class="transition-colors"></div>   <!-- transition-property: color, background-color, border-color, text-decoration-color, fill, stroke; -->
                                        <!-- transition-timing-function: cubic-bezier(0.4, 0, 0.2, 1); -->
                                        <!-- transition-duration: 150ms; -->
<div class="transition-opacity"></div>   <!-- transition-property: opacity; -->
                                         <!-- transition-timing-function: cubic-bezier(0.4, 0, 0.2, 1); -->
                                         <!-- transition-duration: 150ms; -->
<div class="transition-shadow"></div>   <!-- transition-property: box-shadow; -->
                                        <!-- transition-timing-function: cubic-bezier(0.4, 0, 0.2, 1); -->
                                        <!-- transition-duration: 150ms; -->
<div class="transition-transform"></div>   <!-- transition-property: transform; -->
                                            <!-- transition-timing-function: cubic-bezier(0.4, 0, 0.2, 1); -->
                                            <!-- transition-duration: 150ms; -->

<div class="duration-0"></div>   <!-- transition-duration: 0s; -->
<div class="duration-100"></div>   <!-- transition-duration: 100ms; -->
<div class="duration-500"></div>   <!-- transition-duration: 500ms; -->
<div class="duration-1000"></div>   <!-- transition-duration: 1000ms; -->

<div class="ease-linear"></div>   <!-- transition-timing-function: linear -->
<div class="ease-in"></div>   <!-- transition-timing-function: cubic-bezier(0.4, 0, 1, 1); -->
<div class="ease-out"></div>   <!-- transition-timing-function: cubic-bezier(0, 0, 0.2, 1); -->
<div class="ease-in-out"></div>   <!-- transition-timing-function: cubic-bezier(0.4, 0, 0.2, 1); -->

<div class="delay-0"></div>   <!-- transition-delay: 0s; -->
<div class="delay-100"></div>   <!-- transition-delay: 100ms; -->
<div class="delay-500"></div>   <!-- transition-delay: 500ms; -->
<div class="delay-1000"></div>   <!-- transition-delay: 1000ms; -->

<table class="animate-bounce"></table>   <!-- element bounce up and down -->
