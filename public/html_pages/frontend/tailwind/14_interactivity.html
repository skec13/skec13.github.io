Scroll
Syntax
<html class="scroll-auto"></html>   <!-- scroll-behavior: auto; -->
<html class="scroll-smooth"></html>   <!-- scroll-behavior: smooth; -->

<div class="scroll-m-0"></div>   <!-- scroll-margin: 0px; -->
<div class="scroll-mx-0"></div>   <!-- scroll-margin-left: 0px; scroll-margin-right: 0px; -->
<div class="scroll-my-0"></div>   <!-- scroll-margin-top: 0px; scroll-margin-bottom: 0px; -->
<div class="scroll-ms-0"></div>   <!-- scroll-margin-inline-start: 0px; -->
<div class="scroll-me-0"></div>   <!-- scroll-margin-inline-end: 0px; -->
<div class="scroll-mt-0"></div>   <!-- scroll-margin-top: 0px; -->
<div class="scroll-mr-0"></div>   <!-- scroll-margin-right: 0px; -->
<div class="scroll-mb-0"></div>   <!-- scroll-margin-bottom: 0px; -->
<div class="scroll-ml-0"></div>   <!-- scroll-margin-left: 0px; -->

<div class="scroll-p-0"></div>   <!-- scroll-padding: 0px; -->

<!-- snap an element to its center when being scrolled inside a snap container -->
<div class="snap-start"></div>   <!-- scroll-snap-align: start; --> 
<div class="snap-end"></div>   <!-- scroll-snap-align: end; -->    
<div class="snap-center"></div>   <!-- scroll-snap-align: center; -->    
<div class="snap-align-none"></div>   <!-- scroll-snap-align: none; -->    

<!-- to stop on every element when scrolling, use with snap-mandatory on parent div -->
<div class="snap-always"></div>   <!-- scroll-snap-stop: always; -->
<div class="snap-normal"></div>   <!-- scroll-snap-stop: normal; -->

<!-- if stop on an element when scrolling -->
<div class="snap-none"></div>   <!-- dont stop on element -->
<div class="snap-x"></div>   <!-- enable horizontal scroll snapping (use with snap-& on element to work) -->
<div class="snap-y"></div>   <!-- enable vertical scroll snapping -->
<div class="snap-both"></div>   <!-- enable horizontal and vertical scroll snapping -->
<div class="snap-mandatory"></div>   <!-- force a snap container to always come to rest on a snap point -->
<div class="snap-proximity"></div>   <!-- force a snap cointainer to come to rest on snap points in proximity -->
