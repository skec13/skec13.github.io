Sizing
Width and height
Syntax
<div class="w-0"></div>   <!-- width: 0px -->
<div class="w-[100px]"></div>   <!-- width: 100px -->
<div class="min-w-[100px]"></div>   <!-- min-width: 100px -->
<div class="max-w-[100px]"></div>   <!-- max-width: 100px -->

<div class="h-0"></div>   <!-- height: 0px -->
<div class="h-[100px]"></div>   <!-- height: 100px -->
<div class="min-h-[100px]"></div>   <!-- min-height: 100px -->
<div class="max-h-[100px]"></div>   <!-- max-height: 100px -->
