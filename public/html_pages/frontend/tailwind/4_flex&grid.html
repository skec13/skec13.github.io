Flex and grid
Flex
Syntax
<div class="basis-0"></div>   <!-- flex-basis: 0px; -->

<div class="flex-row"></div>
<div class="flex-col"></div>

<div class="flex-wrap"></div>
<div class="flex-nowrap"></div>

<div class="flex-1"></div>   <!-- flex: 1 1 0%; (item can shrink and grow, ignoring its initial size) -->
<div class="flex-auto"></div>   <!-- flex: 1 1 auto; (item can shrink and grow, taking into account its initial size) -->
<div class="flex-initial"></div>   <!-- flex: 0 1 auto; (item can shrink but not grow, taking into account initial size) -->
<div class="flex-none"></div>   <!-- flex: none; (prevents item from shrinking and growing) -->

Grid
Syntax
<div class="grid-cols-1"></div>   <!-- grid-template-columns: repeat(1, minmax(0, 1fr)); -->
<div class="grid-cols-none"></div>   <!-- grid-template-columns: none; -->

<div class="col-span-2"></div>   <!-- grid-column: span 2 / span 2; -->

<div class="grid-rows-1"></div>   <!-- grid-template-rows: repeat(1, minmax(0, 1fr)); -->

<div class="row-span-1"></div>   <!-- grid-row: span 1 / span 1; -->

<div class="grid-flow-row"></div>   <!-- grid-auto-flow: row; -->
<div class="grid-flow-col"></div>   <!-- grid-auto-flow: column; -->
<div class="grid-flow-dense"></div>   <!-- grid-auto-flow: dense; -->
<div class="grid-flow-row-dense"></div>   <!-- grid-auto-flow: row dense; -->
<div class="grid-flow-col-dense"></div>   <!-- grid-auto-flow: column dense; -->

<div class="auto-cols-auto"></div>   <!-- grid-auto-columns: auto; -->
<div class="auto-cols-min"></div>   <!-- grid-auto-columns: min-content; -->
<div class="auto-cols-max"></div>   <!-- grid-auto-columns: max-content; -->
<div class="auto-cols-fr"></div>   <!-- grid-auto-columns: minmax(0, 1fr); -->

<div class="auto-rows-auto"></div>   <!-- grid-auto-rows: auto; -->
<div class="auto-rows-min"></div>   <!-- grid-auto-rows: min-content; -->
<div class="auto-rows-max"></div>   <!-- grid-auto-rows: max-content; -->
<div class="auto-rows-fr"></div>   <!-- grid-auto-rows: minmax(0, 1fr); -->
