Box shadow
Syntax
<div class="shadow-sm"></div>   <!-- box-shadow: 0 1px 2px 0 rgb(0 0 0 / 0.05); -->
<div class="shadow"></div>   <!-- box-shadow: 0 1px 3px 0 rgb(0 0 0 / 0.1), 0 1px 2px -1px rgb(0 0 0 / 0.1); -->
<div class="shadow-md"></div>   <!-- box-shadow: 0 4px 6px -1px rgb(0 0 0 / 0.1), 0 2px 4px -2px rgb(0 0 0 / 0.1); -->
<div class="shadow-lg"></div>   <!-- box-shadow: 0 10px 15px -3px rgb(0 0 0 / 0.1), 0 4px 6px -4px rgb(0 0 0 / 0.1); -->
<div class="shadow-xl"></div>   <!-- box-shadow: 0 20px 25px -5px rgb(0 0 0 / 0.1), 0 8px 10px -6px rgb(0 0 0 / 0.1); -->
<div class="shadow-2xl"></div>   <!-- box-shadow: 0 25px 50px -12px rgb(0 0 0 / 0.25); -->
<div class="shadow-inner"></div>   <!-- box-shadow: inset 0 2px 4px 0 rgb(0 0 0 / 0.05); -->
<div class="shadow-none"></div>   <!-- box-shadow: 0 0 #0000; -->

<div class="shadow-black"></div>   <!-- --tw-shadow-color: #000; -->
