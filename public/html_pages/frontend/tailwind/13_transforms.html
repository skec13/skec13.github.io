Transform classes
Syntax
<div class="scale-0"></div>   <!-- transform: scale(0); -->
                            <div class="scale-50"></div>   <!-- transform: scale(0.5); -->
                            <div class="scale-x-0"></div>   <!-- transform: scaleX(0); -->
                            <div class="scale-y-0"></div>   <!-- transform: scaleY(0); -->

                            <div class="rotate-0"></div>   <!-- transform: rotate(0deg); -->
                            <div class="rotate-45"></div>   <!-- transform: rotate(45deg); -->
                            <div class="rotate-180"></div>   <!-- transform: rotate(180deg); -->

                            <div class="translate-x-0"></div>   <!-- transform: translateX(0px); -->
                            <div class="translate-x-0.5"></div>   <!-- transform: translateX(0.125rem); -->
                            <div class="translate-y-0"></div>   <!-- transform: translateY(0px); -->
                            <div class="translate-y-0.5"></div>   <!-- transform: translateY(0.125rem); -->

                            <div class="skew-x-0"></div>   <!-- transform: skewX(0deg); -->
                            <div class="skew-x-1"></div>   <!-- transform: skewX(1deg); -->
                            <div class="skew-y-0"></div>   <!-- transform: skewY(0deg); -->
                            <div class="skew-y-1"></div>   <!-- transform: skewY(1deg); -->
