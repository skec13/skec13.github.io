Tables
Tables styling
Syntax
table, th, td {
    border: 1px solid;   /*border on all elements*/
    border-collapse: collapse;   /*collapses all borders into one*/
}

td {
    text-align: center;   /*horizontal allignment in cell*/
                left; 
    vertical-align: bottom;   /*vertical allignment*/
}

th, td {
    padding: 15px;
    text-align: left;
    border-bottom: 1px solid #ddd;
    background-color: white;
}
tr:nth-child(even) {background-color: #f2f2f2;}   /*striped table*/

div{
    overflow: auto;   /*div around table to make it responsive*/
}
