                     space-between;
    align-items: center;   /*how to align items on secondary axis*/

                 flex-end;
                 stretch;
                 baseline;
    align-content: space-between;   /*space between flex lines*/

    order: 1;
    flex-grow: 1;   /*defines the ability for a flex item to grow if necessary*/
    flex-shrink: 1;
    flex-basis: 200px;   /*specifies the initial length of the flex item*/
    flex: 50%;   /*takes up 50% of flex container*/
    align-self: center;   /*vertical alignment for specific element*/
                flex-start;
                flex-end;
}
