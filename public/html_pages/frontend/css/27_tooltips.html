Tooltips
Basic tooltip
Syntax
/*tooltip div*/
.tooltip {
    position: relative;
    display: inline-block;
}

/*tooltip text*/
.tooltip .tooltiptext {
    visibility: hidden;
    position: absolute;
    z-index: 1;
    top: -5px;   /*right tooltip*/
    left: 105%;
}

/*show text on hover*/
.tooltip:hover .tooltiptext {
    visibility: visible;
}
Code

<!DOCTYPE >
<html lang="en">
