Align
Center element
Syntax
/*vertical and horizontal center alignment with margins*/  
div {
    margin: 100px auto;                        
}

/*vertical and horizontal center alignment with translate*/
.div-parent {
    position: relative;
}
.div-child {
    position: absolute;
    top: 50%;
    left: 50%;
    transform: translate(-50%, -50%);   /**/
}
Code

<!DOCTYPE >
<html lang="en">
<head>
    <style>
        .first {
            width: 100%;
            border: 2px solid green;
        }
        .first-inner {
            width: 200px;
            height: 100px;
            border: 1px solid black;
            margin: 100px auto;
        }
        .second {
            width: 100%;
            height: 300px;
            border: 2px solid green;
            position: relative;
        }
        .second-inner {
            width: 200px;
            height: 100px;
            border: 2px solid black;
            position: absolute;
            top: 50%;
            left: 50%;
            transform: translate(-50%, -50%);
        }
    </style>
</head>
    <body>
        <div class=first>
            <div class=first-inner>Centered with margins</div>
        </div> 
        <br>
        <div class=second>
            <div class=second-inner>Centered with transform</div>
        </div>  
    </body>
</html>
Output
