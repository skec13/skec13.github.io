Styling input fields
Syntax
input {
    width: 100%;   /*full-width*/
    padding: 12px 20px;   /*for text inside*/
    margin: 8px 0;   /*for space around element*/
    box-sizing: border-box;
}

input[type=text]{}   /*select on text inputs*/

input:focus {}   /*when input is clicked on*/

/*input with icon/image*/
input[type=text] {
    background-color: white;
    background-image: url('searchicon.png');
    background-position: 10px 10px;
    background-repeat: no-repeat;
    padding-left: 40px;
}

/*animated search input*/
input[type=text] {
    transition: width 0.4s ease-in-out;
  }
  
input[type=text]:focus {
    width: 100%;
}

Styling buttons
Syntax
button {
    width: 25%; 
    color: #4CAF50;
    font-size: 20px;
    padding: 10px 10px;
    border-radius: 2px;
    border: 1px solid black;
    box-shadow: 0 8px 16px 0 rgba(0,0,0,0.2), 0 6px 20px 0 rgba(0,0,0,0.19);
    text-align: center;
    text-decoration: none;
    display: inline-block;
    cursor: pointer;
    transition-duration: 0.4s;   /*determine the speed of the hover effect*/
}

button:hover {
    background-color: black;
    color: white;
}

/*horizontal button groups*/
.button {
    float: left;
}

/*vertical button groups*/
.button {
    display: block;
}
