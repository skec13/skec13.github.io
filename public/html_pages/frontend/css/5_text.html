Text properties
Syntax
p{
    color: white   /*color of text*/

    text-align:center;
              :left;
              :justify;

    text-align-last:center;   /*specifies how to align last line of a text*/

    direction:rtl;
    unicode-bidi:bidi-override;   /*changes the direction of the text*/

    writing-mode:vertical-rl;   /*upside down*/

    text-decoration-line:overline;
                        :line-through;
                        :underline;
                        :overline underline;

    text-decoration-color:red;
    text-decoration-style:dotted;
    text-decoration-thickness:5px;

    text-transform:uppercase;
                  :lowercase;
                  :capitalize;

    text-indent:50px;   /*indentation of the first line of a text*/
    letter-spacing:5px;   /*space between characters(-2px) to lower it*/
    line-height:0.8;
    word-spacing:10px;
    white-space:nowrap;   /*wont go in next line*/
               :normal;   /*text will wrap when necessary*/
               :pre;
}

Text overflow
Syntax
p{
    overflow:hidden;
    text-overflow:clip;   /*overflown text is hidden and clipped*/
                 :ellipsis;   /*oveflown text is hidden and ... is shown*/

    word-wrap:break-word;   /*splits word into new line if it overflows*/

    word-break: keep-all;   /*breaks line at hyphens*/ 
    word-break: break-all;   /*breaks line at any character*/
}

<!DOCTYPE >
<html lang="en">
<head>
    <style>
        .first{

Fonts
Syntax
p{
    font-family: Tahoma, Verdana, sans-serif;

    font-style:normal;
              :italic;
              :oblique;

    font-weight:normal;
               :bold;
               :300;

    font-variant:normal;
                :small-caps;

    font-size:40px;
             :2.5em;   /*40px/16=2.5em*/
             :10vw;   /*viewport-1vw=1% of viewport width*/ 
}  

<head>
    <link rel=""stylesheet" href="https://fonts.googleapis.com/css?family=Sofia">   /*google fonts*/
</head>
