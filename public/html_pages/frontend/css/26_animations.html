Animations
Animation properties
Syntax
animation: name duration timing-function delay iteration-count direction fill-mode play-state;

/*animation code*/
@keyframes example {
    from {background-color: red;}
    to {background-color: yellow;}
}

div {
    animation-name: example;
    animation-duration: 4s;
    animation-delay: 1s;
    animation-iteration-count: 3;
                               infinite;
    animation-direction: normal;
                         reverse;
                         alternate;
                         alternate-reverse;
    animation-timing-function: linear;
                               ease;
                               ease-in;
                               ease-out;
                               ease-in-out;
    animation-fill-mode: none;   /*animation will not apply any styles before and after*/
                         forwards;   /*element will retain the style value set by animation*/
                         backwards;   /*element will get the style set by the first keyframe*/
                         both;

}
Code

<!DOCTYPE >
<html lang="en">
<head>
    <style>
        div {
            width: 300px;
            height: 50px;
            background-color: red;
            animation-name: example;
            animation-duration: 4s;
            margin-bottom: 20px;   
        }
        
        @keyframes example {
            from {background-color: red;}
            to {background-color: yellow;}
        }
        .first{
            animation-delay: 2s;
        }
        .second{
            animation-delay: -2s;
        }
        .third{
            animation-iteration-count: 3;
        }
        .fourth{
            animation-direction: reverse;
        }
        .fifth{
            animation-timing-function: ease-in-out;
        }
        .sixth{
            animation-fill-mode: forwards;
        }
    </style>
</head>
<body>
    <div>basic animation</div>
    <div class="first">animation-delay: 2s;</div>
    <div class="second">animation-delay: -2s;</div>
    <div class="third">animation-iteration-count: 3;</div>
    <div class="fourth">animation-direction: reverse;</div>
    <div class="fifth">animation-timing-function: ease-in-out;</div>
    <div class="sixth">animation-fill-mode: forwards;</div>
</body>
</html>
