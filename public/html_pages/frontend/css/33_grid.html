Syntax

    display: grid;   /*takes full width*/
             inline-grid;   /*takes as much space as needed*/
}

grid-template-rows and grid-template-columns property
Syntax
div {
    display: grid;
    grid-template-columns: [first] 40px [second] 50px [third] auto;
                         : 1fr 1fr 1fr;   /*3 equal width columns*/
                         : repeat(auto-fit, minmax(200px, 1fr));   /*best solution*/
    grid-template-rows: [row1-start] 25% [row1-end] 100px [third-line] auto [last-line];
}
