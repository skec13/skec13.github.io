Factorial of a number
Syntax
-> 4! = 4*3*2*1
-> 0! = 1
-> Time complexity: O(n)
Code
function factorial(n){
    let result = 1;
    for(let i = 2; i < n; i++){
        result = result * i;
    }
    return result;
}
Output
Recursive factorial of a number
Syntax
-> 2! = 2*1!
-> 3! = 3*2!
-> n! = n*(n-1)!
-> Time complexity: O(n)
Code
function recursiveFactorial(n){
    if(n === 0){
        return 1;
    }
    return n * recursiveFactorial(n-1);                            
}
