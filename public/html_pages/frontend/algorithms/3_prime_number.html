Prime number
Syntax
-> number greater than 1 that is not a product of two smaller natural numbers
-> Time complexity: O(n)
Code
function isPrime(n){
    if(n < 2){
        return false;
    }   
    for(let i = 2; i < n; i++){
        if(n % i === 0){
            return false;
        }
    }
    return true;
}

Prime number optimization
Syntax
-> integers larger than the square root do not need to be checked
-> Time complexity: O(sqrt(n))
Code
function isPrime(n){
    if(n < 2){
        return false;
    }   
    for(let i = 2; i < Math.sqrt(n); i++){
        if(n % i === 0){
            return false;
        }
    }
    return true;
}
