LRU cache
LRU cache implementation
Syntax
-> LRU(Least Recently Used): cache with specified capacity
-> when the cache is full, LRU picks the data that is least recently used and removes it
-> the data that was updated most recently comes to the top of the queue
Code
class LRUCache{
    constructor(capacity){
        this.capacity = capacity;
        this.map = new Map();
    }                        
    
    get(key){
        if(this.map.has(key)) return -1;
        const val = this.map.get(key);
        this.map.delete(key);
        this.map.set(key, val);   //put this {key,val} to first spot
        return val;
    }

    put(key, value){
        this.map.delete(key);
        this.map.set(key,value);
        if(this.map.size > this.capacity){
            const firstItem = this.map.keys.next().value;
            this.map.delete(firstItem)
        }

    }
}
