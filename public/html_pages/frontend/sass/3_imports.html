Mixin
Syntax
@mixin name {
    property: value;
    property: value;
    ...
}

//mixin usage
@mixin important-text {   //define the mixin
    color: red;
    font-size: 25px;
    font-weight: bold;
    border: 1px solid blue;
}

selector {
    @include mixin-name;   //use the mixin
}

//mixin with variables
@mixin bordered($color, $width) {
    border: $width solid $color;
}

.myArticle {
    @include bordered(blue, 1px);   //call mixin with two variables
}

//Mixin with default value
@mixin bordered($color: blue, $width: 1px) {
    border: $width solid $color;
}
Extend
Syntax
.button-basic {
    border: none;
    padding: 15px 30px;
    text-align: center;
    font-size: 16px;
    cursor: pointer;
}
    
.button-report {
    @extend .button-basic;   //extends all properties of button-basic
    background-color: red;
}
