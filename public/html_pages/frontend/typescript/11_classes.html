Class syntax
Syntax
class User {
    email: string;   //if you dont specify visibility you have to declare it here
    name: string;

    constructor(email: string, name: string){
        this.email: email;
        this.name = name;
    }
}

const user1 = new User("mm@gmail.com", "matevz");
Visibility
Syntax
class Person {
    private name: string;

    public constructor(name: string) {
        this.name = name;
    }

    public getName(): string {
        return this.name;
    }
}
Parameters visibility
Syntax
class Person {
    // if we declare visibility of the variable we dont have to declare it before the constructor
    public constructor(private name: string) {}

    public getName(): string {
        return this.name;
    }
}
Inheritance: implements(define the type a class must follow)
Syntax
interface Shape {
    getArea: () => number;
}

class Rectangle implements Shape {
    public constructor(protected readonly width: number, protected readonly height: number) {}

    public getArea(): number {
        return this.width * this.height;
    }
}
Inheritance: extends
Syntax
class Rectangle {
    //protected properties so the child class can access them
    public constructor(protected readonly width: number, protected readonly height: number) {}

    public getArea(): number {
        return this.width * this.height;
    }
}

class Square extends Rectangle {
    public constructor(width: number){
        super(width, width);
    }

    //getArea gets inherited from Rectangle
}
Getters and setters
Syntax
class User {

    private _count = 1;

    //getter
    get getCount(): number {
        return this._count;
    }

    //setters can not have return type declared in typescript
    set setCount(countNum){
        this._count = countNum;
    }
}
