Objects
Object syntax
Syntax
const car: { type: string, model: string, year: number } = {
    type: "Toyota",
    model: "Corolla",
    year: 2009
};
Function returning an object
Syntax
function createUser(): { name: string, price: number } {
    return { name: "reactjs", price: 399 }
}
Partial property modifier
Syntax
interface Point {
    x: number;
    y: number;
}

let pointPart: Partial<Point> = {}; // `Partial` allows x and y to be optional
Required property modifier
Syntax
interface Car {
    make: string;
    model: string;
    mileage?: number;
}

let myCar: Required<Car> = {
    make: 'Ford',
    model: 'Focus',
    mileage: 12000 // `Required` forces mileage to be defined
};
Record property modifier
Syntax
//specifiers key type and value type of an object
const nameAgeMap: Record<string, number> = {
    'Alice': 21,
    'Bob': 25
};
Omit property modifier
Syntax
interface Person {
    name: string;
    age: number;
    location?: string;
}

const bob: Omit<Person, 'age' | 'location'> = {
    name: 'Bob'
    // `Omit` has removed age and location from the type and they can't be defined here
};
Pick property modifier
Syntax
interface Person {
    name: string;
    age: number;
    location?: string;
}

const bob: Pick<Person, 'name'> = {
    name: 'Bob'
    // Pick removes all but the specified keys from an object type.
};
Exclude property modifier
Syntax
//removes types from a union
type Primitive = string | number | boolean
const value: Exclude<Primitive, string> = true;
Readonly property modifier
Syntax
interface Person {
    name: string;
    age: number;
}
const person: Readonly<Person> = {
    name: "Dylan",
    age: 35,
};
