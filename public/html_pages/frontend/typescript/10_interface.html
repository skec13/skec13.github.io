Interface(alias for objects)
Interface syntax
Syntax
interface Rectangle {
    height: number,
    width: number,
    startTrail(name: string): string
}

const rectangle: Rectangle = {
    height: 20,
    width: 10
};
Extending interface
Syntax
interface Rectangle {
    height: number,
    width: number
}

interface ColoredRectangle extends Rectangle {
    color: string
}

const coloredRectangle: ColoredRectangle = {
    height: 20,
    width: 10,
    color: "red"
};
Adding properties to interface(alias cant do this)
Syntax
interface Window {
    height: number
}

interface Window {
    width: number
}
