Context API(manage state globally)
Using context
Syntax
//context is usually stored in src/store

//src/store/context.jsx
import { createContext } from 'react';

export const CartContext = createContext({
    items: []
});   // creating context object


//App.jsx
import { CartContext } from './stores/context.jsx';

function App() {
    return(
        //wrap all components that need that context
        <CartContext.Provider value={{ items: [] }}>
            <Header></Header>
            <Cart></Cart>
        </CartContext.Provider>
    );
}

export default App;


//Cart.jsx
import { useContext } from 'react';

import { CartContext } from './stores/context.jsx';

export default function Cart() {
    const cartCtx = useContext(CartContext);   //accessing the context values

    return(
        <div>
            {cartCtx.items[1]}
        </div>
    );
}
Linking context to state
Syntax
//App.jsx
import { CartContext } from './stores/context.jsx';

function App() {
    const [shoppingCart, setShoppingCart] = useState({
        items: []
    })

    return(
        //linking state with context
        <CartContext.Provider value={shoppingCart}>
            <Header></Header>
            <Cart></Cart>
        </CartContext.Provider>
    );
}

export default App;
Linking context to state and editing the state with context
Syntax
//src/store/context.jsx
import { createContext } from 'react';

export const CartContext = createContext({
    items: [],
    addItemsToCart: () => {}   //dummy function for autocompletion
});


//App.jsx
import { CartContext } from './stores/context.jsx';

function App() {
    const [shoppingCart, setShoppingCart] = useState({
        items: []
    });

    function handleAddItemToCart() {
        console.log("Added to cart!");
    }

    const ctxValue = {
        items: shoppingCart.items,
        addItemToCart: handleAddItemToCart   //exposing function to context

    }

    return(
        //linking state with context
        <CartContext.Provider value={ctxValue}>
            <Header></Header>
            <Cart></Cart>
        </CartContext.Provider>
    );
}

export default App;
