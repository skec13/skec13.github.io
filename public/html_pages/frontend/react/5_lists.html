export default function Car(props) {
    return <li>I am a { props.brand }</li>;

//App.jsx
import Car from "./Car"

function App() {
    const cars = ['Ford', 'BMW', 'Audi'];
    return (
        <>
        <h1>Who lives in my garage?</h1>
        <ul>
            {cars.map((car) => <Car brand={car} />)}
        </ul>
        </>
    );
}

export default App;

//App.jsx

function App() {
    const cars = ['Ford', 'BMW', 'Audi'];
    return (
        <>
        <h1>Who lives in my garage?</h1>
        <ul>
            {cars.map((car, index) => <li key={index}>{car}</li>)}   //add key prop
        </ul>
        </>
    );
}

export default App;
