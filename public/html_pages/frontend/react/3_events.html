Reacting to events
Syntax
//TabButton.jsx
export default function Button(props){

    function handleClick() {
        console.log("Hello world!");
    }

    return(
        <li>
            <button onClick={handleClick}>{props.children}</button>
        </li>
    );
}

//App.jsx

import TabButton from './components/TabButton';

function App() {
  return (
    <div className="App">
      <TabButton>Home</TabButton>
      <TabButton>About</TabButton>
      <TabButton>Contacts</TabButton>
    </div>
  );
}

export default App;
Passing functions as values to props
Syntax
//TabButton.jsx
export default function TabButton(props){
    return(
    <li>
        <button onClick={props.onSelect}>{props.children}</button>
    </li>       
    );
}

//App.jsx
import TabButton from './components/TabButton';

function App() {
    function handleClick(){
      alert("passed as value");
    }
  
    return (
      <div className="App">
        <TabButton onSelect={handleClick}>Home</TabButton>
      </div>
    );
  }
  
export default App;
Passing custom arguments to event functions
Syntax
//TabButton.jsx
export default function TabButton1(props){
    function handleClick(content){
        alert(content);
    }

    return(
    <li>
        <button onClick={() => handleClick(props.children)}>{props.children}</button>
    </li>       
    );
}

//App.jsx
import TabButton from './components/TabButton';

function App() {
  return (
    <div className="App">
      <TabButton>Home</TabButton>
    </div>
  );
}

export default App;
