Syntax
//Default parameters
function myFunction(x, y = 10) {
    return x + y;
}

//Rest parameters
function sum(...args) {
    let sum = 0;
    for (let arg of args) sum += arg;
    return sum;
}
    
let x = sum(4, 9, 16, 25, 29, 100, 66, 77);

//Argument object(stores all argument of when the function was called)
x = sumAll(1, 123, 500, 115, 44, 88);

function sumAll() {
let sum = 0;
for (let i = 0; i < arguments.length; i++) {
    sum += arguments[i];
}
    return sum;
}

call() method (an object can use a method belonging to another object)
Syntax
const person = {
    fullName: function() {
    return this.firstName + " " + this.lastName;
    }
}
const person1 = {
    firstName:"John",
    lastName: "Doe"
}

// This will return "John Doe":
person.fullName.call(person1);   //uses fullName method on person1 object

bind() method (object can borrow a method from another object)
Syntax
const person = {
    firstName:"John",
    lastName: "Doe",
    fullName: function () {
        return this.firstName + " " + this.lastName;
    }
}

const member = {
    firstName:"Hege",
    lastName: "Nilsen",
}

let fullName = person.fullName.bind(member);
