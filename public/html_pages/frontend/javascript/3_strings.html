String methods
Syntax
string.length;   //returns the length of the string
string.slice(start, end);   //returns new substring of original string
string.substring(start, end);   //similar to slice-values less than 0 are treated as 0
string.repeat(count);   //returns new string which contains the specified number of copies
string.replace(pattern, replacement);   //replaces a specified value in string-returns new string
string.replaceAll(pattern, replacement);   //introduced in 2021 to replace regular expressions
string.toUpperCase();
string.startsWith(searchString, position);   //returns boolean
string.concat(string1, string2);   //joins two or more strings
string.trim();   //removes whitespace from both sides of a string and returns new string
string.trimStart();   //removes whitespace from the beginning of this string and returns a new string
string.trimEnd();   //removes whitespace from the end of the string and returns a new string
string.padStart(targetLength, padString);   //adds number of padString characters before string
string.padEnd(targetLength, padString);
string.charAt(index);   //returns the character at given index
string.charCodeAt(index);   //returns the unicode at given index
string.split(separator, limit);   //coverts string to an array
string.toString();   //returns string value

Syntax
string.indexOf(searchString, position);   //searches this string and returns index of the first occurence
string.lastIndexOf(searchString, position);   //searches this string and returns index of the last occurence
string.search(regexp);   //executes the search with regular expression
string.match(regexp);   //retrieves the result of matching this string against a regular expression
string.matchAll(regexp);   //returns an iterator of all results matching this string against a regular expression
string.includes(searchString, position);   //case-sensitive search to determine whether a given string may be found within this string
