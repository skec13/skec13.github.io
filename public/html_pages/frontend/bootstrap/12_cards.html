Card classes
Syntax
<div class="card">
    <div class="card-body">Basic card</div>   <!-- basic card -->
</div>

<div class="card">
    <div class="card-header">Header</div>   <!-- card header -->
    <div class="card-body">Content</div>   <!-- card body -->
    <div class="card-footer">Footer</div>   <!-- card footer -->
</div>

<div class="card">
    <div class="card-body">
        <h4 class="card-title">Card title</h4>   <!-- card title -->
        <p class="card-text">Some example text. Some example text.</p>   <!-- card text -->
        <a href="#" class="card-link">Card link</a>   <!-- card link -->
        <a href="#" class="card-link">Another link</a>
    </div>
</div>

<div class="card" style="width:400px">
    <img class="card-img-top" src="img_avatar1.png" alt="Card image">   <!-- card image -->
    <div class="card-body">
        <h4 class="card-title">John Doe</h4>
        <p class="card-text">Some example text.</p>
        <a href="#" class="btn btn-primary">See Profile</a>
    </div>
</div>
