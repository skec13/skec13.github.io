Modal
Modal classes
Syntax
<button type="button" class="btn btn-primary" data-bs-toggle="modal" data-bs-target="#myModal">   <!-- button to open modal -->
    Open modal
</button>

<div class="modal" id="myModal"></div>   <!-- modal -->
<div class="modal fade"></div>   <!-- modal with animation -->

<div class="modal-dialog"></div>   <!-- modal window -->
<div class="modal-dialog modal-sm"></div>   <!-- small modal window -->
<div class="modal-dialog modal-fullscreen"></div>   <!-- fullscreen modal -->
<div class="modal-dialog modal-fullscreen-sm-down"></div>   <!-- fullscreen below 576px -->
<div class="modal-dialog modal-dialog-centered"></div>   <!-- centered modal -->
<div class="modal-dialog modal-dialog-scrollable"></div>   <!-- only modal scrollable if content too big(not whole page) -->

<div class="modal-content"></div>   <!-- modal content -->

<div class="modal-header">   <!-- modal header -->
    <h4 class="modal-title">Modal Heading</h4>
    <button type="button" class="btn-close" data-bs-dismiss="modal"></button>
</div>

<div class="modal-body">   <!-- modal body -->
    Modal body..
</div>

<div class="modal-footer">   <!-- modal footer -->
    <button type="button" class="btn btn-danger" data-bs-dismiss="modal">Close</button>
</div>
