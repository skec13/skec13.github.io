Progress bar classes
Syntax
<ul class="pagination"></ul>    <!-- basic pagination -->
<ul class="pagination pagination-lg"></ul>    <!-- large pagination -->
<ul class="pagination pagination-sm"></ul>    <!-- small pagination -->
<ul class="pagination justify-content-center"></ul>    <!-- centered pagination -->
<ul class="pagination" justify-content-end></ul>    <!-- right aligned pagination -->
<ul class="breadcrumb"></ul>    <!-- basic breadcrumb(underlined links) -->

<li class="page-item"><a class="page-link" href="#">Previous</a></li>   <!-- basic item -->
<li class="page-item active"><a class="page-link" href="#">2</a></li>   <!-- active item -->
<li class="page-item disabled"><a class="page-link" href="#">Previous</a></li>   <!-- disabled item -->
