Border classes
Syntax
.border   <!-- border -->
.border-1   <!-- border width 1 -->
.border-2   <!-- border  width 2 -->
.border-3   <!-- border width 3 -->
.border-4   <!-- border width 4 -->
.border-5   <!-- border width 5 -->
.border-0   <!-- remove border -->
.border-top-0   <!-- remove top border -->
.border-top   <!-- only top border -->
.border-end   <!-- only right border -->
.border-bottom   <!-- only bottom border -->
.border-start   <!-- only left border -->
.border-primary   <!-- blue border -->

.rounded   <!-- rounded corners -->
.rounded-1   <!-- rounded corners 1 -->
.rounded-2   <!-- rounded corners 2 -->
.rounded-3   <!-- rounded corners 3 -->
.rounded-4   <!-- rounded corners 4 -->
.rounded-5   <!-- rounded corners 5 -->
.rounded-0   <!-- remove rounded corners -->
.rounded-top   <!-- only top rounded corners -->
.rounded-circle   <!-- circle -->
.rounded-pill  <!-- pill (add width) -->

Spacing classes
Syntax
.m-#   <!-- margin on all sides -->
.mt-#   <!-- margin top -->
.mb-#   <!-- margin bottom -->
.ms-#   <!-- margin left -->
.me-#   <!-- margin right -->
.mx-#   <!-- margin left and right -->
.mx-auto   <!-- center align -->
.my-#   <!-- margin top and bottom -->

.p-#   <!-- padding on all sides -->
