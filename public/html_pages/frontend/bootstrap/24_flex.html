Flex
Flex classes
Syntax
<!-- flex container -->
<div class="d-flex"></div>   <!-- flex container -->
<div class="d-inline-flex"></div>   <!-- inline flex container -->

<div class="d-flex flex-row"></div>   <!-- flex direction -->
<div class="d-flex flex-column"></div>   <!-- flex direction -->

<div class="d-flex justify-content-start"></div>   <!-- justify content(horizontal allignment of content inside flex container) -->
<div class="d-flex align-content-start"></div>   <!-- align content(vertical allignment of content inside flex container) -->
<div class="d-flex align-items-start"></div>   <!-- align items(vertical allignment of items in single row) -->

<div class="d-flex flex-wrap"></div>   <!-- wrap content -->

<!-- flex item -->
<div class="flex-fill">Flex item 1</div>   
<div class="flex-fill">Flex item 2</div>   
<div class="flex-fill">Flex item 3</div>   <!-- 3 equal width items -->

<div class="order-1">Flex item 3</div>   <!-- order of item -->

<div class="flex-grow-1">Flex item 3</div>   <!-- grow of item -->

<div class="me-auto">Flex item 3</div>   <!-- margin end (pushes item to the left) -->
