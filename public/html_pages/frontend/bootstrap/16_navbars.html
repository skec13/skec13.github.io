Navbars
Navbar classes
Syntax
<nav class="navbar navbar-expand-sm bg-light"></nav>   <!-- grey horizontal navbar(vertical on small screens) -->
<nav class="navbar bg-light"></nav>   <!-- always vertical navbar -->
<nav class="navbar navbar-expand-sm bg-light justify-content-center"></nav>   <!-- centered navbar -->
<nav class="navbar navbar-expand-sm bg-dark navbar-dark fixed-top"></nav>   <!-- fixed top navbar -->
<nav class="navbar navbar-expand-sm bg-dark navbar-dark fixed-bottom"></nav>   <!-- fixed bottom navbar -->
<nav class="navbar navbar-expand-sm bg-dark navbar-dark sticky-top"></nav>   <!-- sticky-top navbar -->

<div class="container-fluid">   <!-- navbar container -->
    <ul class="navbar-nav">
        <li class="nav-item">
            <a class="nav-link" href="#">Link 1</a>   <!-- anchor item -->
        </li>
        <li class="nav-item">
            <a class="nav-link" href="#">Link 2</a>
        </li>
        <li class="nav-item">
            <a class="nav-link" href="#">Link 3</a>
        </li>
    </ul>
</div>

<div class="container-fluid">
    <a class="navbar-brand" href="#">Logo</a>   <!-- logo item -->
</div>

<div class="container-fluid">
    <a class="navbar-brand" href="#">
      <img src="logo.png" alt="Avatar Logo" style="width:40px;" class="rounded-pill">   <!-- image item -->
    </a>
</div>

<nav class="navbar navbar-expand-sm bg-dark navbar-dark">
    <div class="container-fluid">
      <span class="navbar-text">Navbar text</span>   <!-- text item -->
    </div>
</nav>
