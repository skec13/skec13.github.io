Images
Image classes
Syntax
<img src="cinqueterre.jpg" class="rounded" alt="Cinque Terre">   <!-- rounded image -->
<img src="cinqueterre.jpg" class="rounded-circle" alt="Cinque Terre">   <!-- cirlce image -->
<img src="cinqueterre.jpg" class="img-thumbnail" alt="Cinque Terre">   <!-- bordered image -->
<img src="cinqueterre.jpg" class="float-start" alt="Cinque Terre">   <!-- left-aligned image -->
<img src="cinqueterre.jpg" class="float-end" alt="Cinque Terre">   <!-- right-aligned image -->
<img src="cinqueterre.jpg" class="mx-auto d-block" alt="Cinque Terre">   <!-- centered image with margin:auto and display:block -->
<img src="cinqueterre.jpg" class="img-fluid" alt="Cinque Terre">   <!-- responsive image -->

<!DOCTYPE >
<html lang="en">
<head>
    <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/css/bootstrap.min.css" rel="stylesheet">
    <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/js/bootstrap.bundle.min.js"></script>
    <meta name="viewport" content="width=device-width, initial-scale=1">
</head>
<body>
    <p>Rounded image:</p>
    <img src="img.png" class="rounded" alt="Cinque Terre" width="300" height="200">
    <br>
    <p>Rounded-circle image:</p>
    <img src="img.png" class="rounded-circle" alt="Cinque Terre" width="300" height="200">
    <br>
    <p>Thumbnail image:</p>
    <img src="img.png" class="img-thumbnail" alt="Cinque Terre" width="300" height="200">
    <br>
    <p>Left-aligned image:</p>
    <img src="img.png" class="float-start" alt="Cinque Terre" width="300" height="200">
    <br>
    <p>Right-aligned image:</p>
    <img src="img.png" class="float-end" alt="Cinque Terre" width="300" height="200">
    <br>
    <p>Centered image:</p>
    <img src="img.png" class="mx-auto d-block" alt="Cinque Terre" width="300" height="200">
    <br>
    <p>Responsive image(scales to parent max-width: 100% and height: auto):</p>
    <img src="img.png" class="img-fluid" alt="Cinque Terre">
</body>
</html>
