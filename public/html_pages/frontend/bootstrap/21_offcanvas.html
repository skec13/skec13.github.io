Offcanvas
Offcanvas as side menu
Syntax
<div class="offcanvas offcanvas-start" id="demo"></div>   <!-- offcanvas element -->
<div class="offcanvas offcanvas-end" id="demo"></div>   <!-- right offcanvas element -->
<div class="offcanvas offcanvas-top" id="demo"></div>   <!-- top offcanvas element -->
<div class="offcanvas offcanvas-bottom" id="demo"></div>   <!-- bottom offcanvas element -->
<div class="offcanvas offcanvas-start offcanvas-lg" id="demo"></div>   <!-- responsive offcanvas -->

<div class="offcanvas-header">   <!-- offcanvas header -->
    <h1 class="offcanvas-title">Heading</h1>   <!-- offcanvas title -->
    <button type="button" class="btn-close text-reset" data-bs-dismiss="offcanvas"></button>   <!-- closing button -->
</div>

<div class="offcanvas-body">   <!-- offcanvas body -->
    <p>Some text lorem ipsum.</p>
    <button class="btn btn-secondary" type="button">A Button</button>
</div>


<button class="btn btn-primary" type="button" data-bs-toggle="offcanvas" data-bs-target="#demo">   <!-- button for opening offcanvas -->
    Open Offcanvas Sidebar
</button>
