Containers
Getting started and containers
Syntax
<head>
    <!-- Latest compiled and minified CSS -->
    <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/css/bootstrap.min.css" rel="stylesheet">
    <!-- Latest compiled JavaScript -->
    <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/js/bootstrap.bundle.min.js"></script>
    /*bootstrap is mobile first-include this to ensure proper rendering and touch zooming*/
    <meta name="viewport" content="width=device-width, initial-scale=1">
</head>

<!-- fixed width container with breakpoint at small:576px, medium:768px, large:992px, x-large:1200px, xx-large:1400px -->
<div class="container"></div>

<!-- fluid container of full width -->
<div class="container"></div>

<!-- responsive containers -->
<div class="container-sm"></div>   100% wide until small breakpoint
<div class="container-md"></div>   100% wide until medium breakpoint
<div class="container-lg"></div>   100% wide until large breakpoint
<div class="container-xl"></div>   100% wide until extra large breakpoint
<div class="container-xxl"></div>   100% wide until extra extra large breakpoint
