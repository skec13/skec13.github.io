Syntax
<ul class="nav"></ul>   <!-- basic nav -->
<ul class="nav justify-content-center"></ul>   <!-- centered nav -->
<ul class="nav justify-content-end"></ul>   <!-- right aligned nav -->
<ul class="nav flex-column"></ul>   <!-- vertical nav -->
<ul class="nav nav-tabs"></ul>   <!-- tabs nav -->
<ul class="nav nav-pills"></ul>   <!-- pills nav -->
<ul class="nav nav-pills nav-justified"></ul>   <!-- pills nav with spacing -->

<li class="nav-item">
    <a class="nav-link" href="#">Link</a>   <!-- basic nav item -->
</li>

<li class="nav-item">

</li>
