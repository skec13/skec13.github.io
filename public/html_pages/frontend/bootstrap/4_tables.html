Tables
Table classes
Syntax
<table class="table"></table>   <!-- basic table with padding and horizontal dividers -->
<table class="table table-striped"></table>   <!-- adds zebra stripes -->
<table class="table table-bordered"></table>   <!-- adds borders -->
<table class="table table-hover"></table>   <!-- adds a hover effect -->
<table class="table table-dark"></table>   <!-- adds a black background -->
<table class="table table-dark table-stripped"></table>   <!-- adds dark theme and zebra stripes -->
<table class="table table-borderedless"></table>   <!-- adds borders -->

<tr class="table-primary"></tr>   <!-- paint table row blue -->

<table class="table table-sm"></table>   <!-- smaller table -->
<table class="table table-responsive"></table>   <!-- adds horizontal scroll if needed -->
<table class="table table-responsive-md"></table>   <!-- adds horizontal scroll if bigger than 768px -->
