Grid
grid classes
Syntax
<div class="row"></div>   <!-- grid div -->
<div class="row row-cols-1"></div>   <!-- grid div with only 1 column(full width) -->
<div class="row g-0"></div>   <!-- remove extra space between columns -->

<div class="col-*-*"></div>   <!-- grid item, first * is responsiveness, second * is number which should add up to 12 or fewer -->

<div class="col"></div>   <!-- horizontall at all times -->
<div class="col-sm"></div>   <!-- collapsed to start, horizontal above 576px -->
<div class="col-md"></div>   <!-- collapsed to start, horizontal above 768px -->
<div class="col-lg"></div>   <!-- collapsed to start, horizontal above 992px -->
<div class="col-xl"></div>   <!-- collapsed to start, horizontal above 1200px -->
<div class="col-xll"></div>   <!-- collapsed to start, horizontal above 1400px -->
