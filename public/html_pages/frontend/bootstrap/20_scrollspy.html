Scrollspy
Scrollspy classes
Syntax
<body data-bs-spy="scroll" data-bs-target=".navbar" data-bs-offset="50"></body>   <!-- scrollable area with target(navbar) -->

<nav class="navbar navbar-expand-sm bg-dark navbar-dark fixed-top">   <!-- target of scrollable area -->
    <ul class="navbar-nav">
        <li><a href="#section1">Section 1</a></li>
    </ul>
</nav>

<div id="section1">   <!-- id of section must match href of anchor -->
    <h1>Section 1</h1>
    <p>Try to scroll this page</p>
</div>
