Buttons
Button classes
Syntax
<button type="button" class="btn">Basic</button>   <!-- basic button -->
<button type="button" class="btn btn-primary">Primary</button>   <!-- blue button -->
<button type="button" class="btn btn-secondary">Secondary</button>   <!-- grey button -->
<button type="button" class="btn btn-success">Success</button>   <!-- green button -->
<button type="button" class="btn btn-info">Info</button>   <!-- light blue button -->
<button type="button" class="btn btn-warning">Warning</button>   <!-- yellow button -->
<button type="button" class="btn btn-danger">Danger</button>   <!-- red button -->
<button type="button" class="btn btn-dark">Dark</button>   <!-- black button -->
<button type="button" class="btn btn-light">Light</button>   <!-- white button -->
<button type="button" class="btn btn-link">Link</button>   <!-- link button -->

<!-- button classes can be used on <a> <button> and <input> -->

<button type="button" class="btn btn-outline-primary">Primary</button>   <!-- outline button -->

<button type="button" class="btn btn-primary btn-lg"></button>   <!-- large button -->
<button type="button" class="btn btn-primary"></button>   <!-- regular button -->
<button type="button" class="btn btn-primary btn-sm"></button>   <!-- small button -->

<div class="d-grid">
    <button type="button" class="btn btn-primary btn-block"></button>   <!-- full-width button -->
</div>

<div class="d-grid gap-3">
    <button type="button" class="btn btn-primary btn-block"></button>
    <button type="button" class="btn btn-primary btn-block"></button>
    <button type="button" class="btn btn-primary btn-block"></button>   <!-- 3 full width buttons with gap -->
</div>

<button type="button" class="btn btn-primary active"></button>   <!-- active button -->
<button type="button" class="btn btn-primary" disabled></button>   <!-- disable button -->

<button class="btn btn-primary">
    <span class="spinner-border spinner-border-sm"></span>   <!-- loading button -->
</button>
  
<button class="btn btn-primary">
    <span class="spinner-border spinner-border-sm"></span>   <!-- loading button -->
    Loading..
</button>
  
<button class="btn btn-primary" disabled>
    <span class="spinner-border spinner-border-sm"></span>   <!-- loading button -->
    Loading..
</button>
  
<button class="btn btn-primary" disabled>
    <span class="spinner-grow spinner-grow-sm"></span>   <!-- loading button -->
    Loading..
</button>

Button groups
Syntax
<div class="btn-group">   <!-- horizontal button group -->
    <button type="button" class="btn btn-primary">Apple</button>
    <button type="button" class="btn btn-primary">Samsung</button>
    <button type="button" class="btn btn-primary">Sony</button>
</div>

<div class="btn-group-vertical">   <!-- vertical button group -->
    <button type="button" class="btn btn-primary">Apple</button>
    <button type="button" class="btn btn-primary">Samsung</button>
    <button type="button" class="btn btn-primary">Sony</button>
</div>

<div class="btn-group">   <!-- nested button group -->
    <button type="button" class="btn btn-primary">Apple</button>
    <button type="button" class="btn btn-primary">Samsung</button>
    <div class="btn-group">
        <button type="button" class="btn btn-primary dropdown-toggle" data-bs-toggle="dropdown">Sony</button>
        <div class="dropdown-menu">
            <a class="dropdown-item" href="#">Tablet</a>
            <a class="dropdown-item" href="#">Smartphone</a>
        </div>
    </div>
</div>
