Input classes
Syntax
<input type="text" class="form-control">   <!-- all inputs with .form-control get styling -->
<input type="password" class="form-control">   <!-- password input -->
<input type="checkbox" class="form-control">   <!-- checkbox -->
<textarea class="form-control"></textarea>   <!-- textarea -->

<input type="text" class="form-control form-control-lg">   <!-- large input -->
<input type="text" class="form-control">   <!-- normal input -->
<input type="text" class="form-control form-control-sm">   <!-- small input -->

<input type="text" class="form-control" disabled>   <!-- disabled input -->
<input type="text" class="form-control" readonly>   <!-- readonly input -->

<input type="text" class="form-control-plaintext">   <!-- input without borders -->

<input type="Colors" class="form-control form-control-color">   <!-- color input -->
<input type="range" class="form-range" min="0" max="4">   <!-- range input -->
