List groups
List groups classes
Syntax
<ul class="list-group"></ul>    <!-- basic list group -->
<ul class="list-group list-group-flush"></ul>    <!-- list group without borders -->
<ul class="list-group list-group-numbered"></ul>    <!-- numbered list group -->
<ul class="list-group list-group-horizontal"></ul>    <!-- horizontal list group -->

<li class="list-group-item">First item</li>   <!-- basic group item -->
<li class="list-group-item active">First item</li>   <!-- active group item -->
<a href="#" class="list-group-item list-group-item-action">First item</a>   <!-- anchor group item -->
<li class="list-group-item disabled">First item</li>   <!-- disabled group item -->
<li class="list-group-item list-group-item-success">First item</li>   <!-- green group item -->
<li class="list-group-item d-flex justify-content-between align-items-center">   <!-- group item with badges -->
    Inbox
    <span class="badge bg-primary rounded-pill">12</span>
</li>
