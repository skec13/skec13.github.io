Tooltips
Tooltip classes
Syntax
<p  data-bs-toggle="tooltip">Hover over me for tooltip</p>   <!-- adding tooltip to element -->
<p  data-bs-toggle="tooltip" data-bs-placement="top" >Hover over me for tooltip</p>   <!-- top tooltip -->
<p  data-bs-toggle="tooltip" data-bs-placement="bottom" >Hover over me for tooltip</p>   <!-- bottom tooltip -->
<p  data-bs-toggle="tooltip" data-bs-placement="left" >Hover over me for tooltip</p>   <!-- left tooltip -->
<p  data-bs-toggle="tooltip" data-bs-placement="right" >Hover over me for tooltip</p>   <!-- right tooltip -->

<!-- enabling tooltips in javascript -->
<script>
    var tooltipTriggerList = [].slice.call(document.querySelectorAll('[data-bs-toggle="tooltip"]'));
    var tooltipList = tooltipTriggerList.map(function (tooltipTriggerEl) {
        return new bootstrap.Tooltip(tooltipTriggerEl);
    })
</script>
Code

<!DOCTYPE >
<html lang="en">
<head>
    <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/css/bootstrap.min.css" rel="stylesheet">
    <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/js/bootstrap.bundle.min.js"></script>
    <meta name="viewport" content="width=device-width, initial-scale=1">
</head>
<body>
    <p data-bs-toggle="tooltip">Hover over me!</p>
    <script>
        var tooltipTriggerList = [].slice.call(document.querySelectorAll("[data-bs-toggle="tooltip"]"))
        var tooltipList = tooltipTriggerList.map(function (tooltipTriggerEl) {
        return new bootstrap.Tooltip(tooltipTriggerEl)
        })
    </script>
</body>
</html>

Popover classes
Syntax
<button type="button" class="btn btn-primary" data-bs-toggle="popover" title="Popover Header" data-bs-content="Some content inside the popover">
    Toggle popover
</button>   <!-- basic popover -->

<script>
    var popoverTriggerList = [].slice.call(document.querySelectorAll('[data-bs-toggle="popover"]'))
    var popoverList = popoverTriggerList.map(function (popoverTriggerEl) {
      return new bootstrap.Popover(popoverTriggerEl)
    })
</script>

<a href="#" title="Header" data-bs-toggle="popover" data-bs-placement="top" data-content="Content">Top</a>   <!-- top popover -->

<a href="#" title="Dismissible popover" data-bs-toggle="popover" data-bs-trigger="focus" data-bs-content="Closing anywhere">
    Click me
</a>   <!-- closing popover when clicking outside the element -->   

<a href="#" title="Header" data-bs-toggle="popover" data-bs-trigger="hover" data-bs-content="Popover text">Hover over me</a>   <!-- hoverable popover -->
