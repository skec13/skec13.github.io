Typography
Text styling
Syntax
<p class="h1"></p>   <!-- show p element as h1 -->

<h1 class="display-1"></h1>   <!-- display-1 to display-6 to make text bigger -->

<p class="lead"></p>   <!-- makes a paragraph stand out -->
<p class="text-start"></p>   <!-- left-aligned text -->
<p class="text-break"></p>   <!-- prevents long text from breaking layout -->
<p class="text-center"></p>   <!-- center-aligned text -->
<a class="text-decoration-none"></a>   <!-- removes the underline from a link -->
<p class="text-end"></p>   <!-- right-aligned text -->
<p class="text-no-wrap"></p>   <!-- no-wrap text -->
<p class="text-lowercase"></p>   <!-- lowercased text -->
<p class="text-uppercase"></p>   <!-- uppercased text -->
<p class="text-capitalize"></p>   <!-- capitalized text -->
<ul class="list-unstyled"></ul>   <!-- removes the default list-stlye and left margin -->
<ul class="list-inline"></ul>   <!-- places all list items on a single line -->
